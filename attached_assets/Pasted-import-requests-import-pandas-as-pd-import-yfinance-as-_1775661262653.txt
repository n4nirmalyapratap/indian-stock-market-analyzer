import requests
import pandas as pd
import yfinance as yf
import json
import io
import time
from datetime import datetime
from zipfile import ZipFile
from concurrent.futures import ThreadPoolExecutor, as_completed

def download_amfi_classification():
    """Downloads the latest semi-annual stock classification list from the AMFI website."""
    print("Attempting to download official AMFI stock classification list...")
    base_url = "https://www.amfiindia.com/Themes/Theme1/downloads/Updated-list-of-large-mid-small-cap-{}.xlsx"
    now = datetime.now()
    year = now.year
    
    # Determine the most likely current period string (e.g., 30-Jun-2023 or 31-Dec-2022)
    # Note: This URL pattern may change and might require manual updates if the script fails.
    if now.month >= 7:
        period_str = f"30-Jun-{year}"
    else:
        period_str = f"31-Dec-{year - 1}"
    
    file_url = base_url.format(period_str)
    
    try:
        response = requests.get(file_url)
        response.raise_for_status()
        with pd.ExcelFile(io.BytesIO(response.content)) as xls:
            df = pd.read_excel(xls, header=1) # Data starts from the second row
        print(f"Successfully downloaded and read AMFI list for period: {period_str}")
        df.rename(columns={'Ticker': 'SYMBOL', 'Category ': 'Category'}, inplace=True)
        # The AMFI list uses NSE symbols in the 'Ticker' column
        amfi_map = pd.Series(df.Category.values, index=df.SYMBOL).to_dict()
        return amfi_map
    except requests.exceptions.RequestException as e:
        print(f"Could not download AMFI list for {period_str}: {e}. Check AMFI website for the correct filename.")
        return {}

def get_bse_stocks():
    """Downloads and reads the complete BSE equity list from the daily Bhavcopy file."""
    print("Downloading complete list of BSE securities from Bhavcopy...")
    today = datetime.now()
    date_str = today.strftime("%d%m%y")
    url = f"https://www.bseindia.com/download/BhavCopy/Equity/EQ_ISINCODE_{date_str}.zip"
    
    try:
        headers = {'User-Agent': 'Mozilla/5.0'}
        response = requests.get(url, headers=headers)
        response.raise_for_status()
        with ZipFile(io.BytesIO(response.content)) as z:
            csv_filename = f"EQ_ISINCODE_{date_str}.CSV"
            with z.open(csv_filename) as f:
                df = pd.read_csv(f)
                df = df[['SC_CODE', 'SC_NAME']].copy()
                df['SC_CODE'] = df['SC_CODE'].astype(str)
                print(f"Successfully parsed {len(df)} BSE securities.")
                return df
    except Exception as e:
        print(f"Error processing BSE Bhavcopy file (it may not be generated yet for today or it's a holiday): {e}")
        return pd.DataFrame()

def get_nse_stocks():
    """Downloads the complete list of NSE securities."""
    print("Downloading complete list of NSE securities...")
    nse_url = "https://nsearchives.nseindia.com/content/equities/EQUITY_L.csv"
    headers = {'User-Agent': 'Mozilla/5.0'}
    try:
        response = requests.get(nse_url, headers=headers)
        response.raise_for_status()
        df = pd.read_csv(io.BytesIO(response.content))
        print(f"Successfully downloaded {len(df)} NSE securities.")
        return df[['SYMBOL', 'NAME OF COMPANY']]
    except Exception as e:
        print(f"An error occurred downloading the NSE list: {e}")
        return pd.DataFrame()

def get_market_cap_category(market_cap_inr):
    """
    Fallback categorizer based on market capitalization in INR.
    Uses commonly accepted thresholds for the Indian market.
    Large-Cap: > 2,00,000 Cr
    Mid-Cap: 50,000 Cr to 2,00,000 Cr
    Small-Cap: 500 Cr to 50,000 Cr
    Micro-Cap: < 500 Cr [...](asc_slot://start-slot-60)
    """
    if market_cap_inr is None: return "Unknown"
    if market_cap_inr >= 200000000000: return "Large-Cap"
    elif 50000000000 <= market_cap_inr < 200000000000: return "Mid-Cap"
    elif 500000000 <= market_cap_inr < 50000000000: return "Small-Cap"
    else: return "Micro-Cap"

def get_stock_info(ticker):
    """Retrieves market cap and long name for a single stock ticker, handling errors gracefully."""
    try:
        stock = yf.Ticker(ticker)
        info = stock.info
        market_cap = info.get('marketCap')
        long_name = info.get('longName')
        # yfinance can return empty dicts or missing keys for failed tickers
        if not market_cap:
            return ticker, None
        return ticker, {'marketCap': market_cap, 'longName': long_name}
    except Exception as e:
        # Log errors for failed tickers for later review
        # print(f"Could not fetch info for {ticker}: {e}")
        return ticker, None

def generate_stock_json():
    """Fetches all stocks, categorizes them, and saves to a JSON file."""
    amfi_categories = download_amfi_classification()
    nse_stocks = get_nse_stocks()
    bse_stocks = get_bse_stocks()

    # --- Prepare Tickers for Parallel Fetch ---
    nse_tickers = [f"{symbol}.NS" for symbol in nse_stocks['SYMBOL']]
    bse_tickers = [f"{code}.BO" for code in bse_stocks['SC_CODE']]
    all_tickers = list(set(nse_tickers + bse_tickers))
    print(f"\nPreparing to fetch data for {len(all_tickers)} unique tickers...")

    # --- Fetch Data in Parallel with Rate Limiting ---
    live_stock_data = {}
    with ThreadPoolExecutor(max_workers=10) as executor:
        futures = {executor.submit(get_stock_info, ticker): ticker for ticker in all_tickers}
        for i, future in enumerate(as_completed(futures)):
            ticker, data = future.result()
            if data:
                live_stock_data[ticker] = data
            print(f"Processed {i + 1}/{len(all_tickers)}: {ticker}")
            # Introduce a delay after every 100 requests to avoid rate limiting [...](asc_slot://start-slot-62)
            if (i + 1) % 100 == 0:
                print("Pausing for 5 seconds to respect API rate limits...")
                time.sleep(5)

    # --- Consolidate and Categorize ---
    final_json_data = []
    processed_symbols = set()
    
    # Process NSE stocks
    for _, row in nse_stocks.iterrows():
        symbol = row['SYMBOL']
        if symbol in processed_symbols: continue
        
        ticker = f"{symbol}.NS"
        info = live_stock_data.get(ticker)
        if info and info.get('marketCap'):
            # 1. Use official AMFI category if available
            category = amfi_categories.get(symbol, None)
            # 2. Fallback to live market cap if not in AMFI list
            if not category:
                category = get_market_cap_category(info['marketCap'])
            
            final_json_data.append({
                "symbol": symbol,
                "company_name": info.get('longName', row['NAME OF COMPANY']),
                "exchange": "NSE",
                "market_cap_inr": info['marketCap'],
                "category": category
            })
            processed_symbols.add(symbol)

    # Process BSE stocks
    for _, row in bse_stocks.iterrows():
        symbol = row['SC_CODE']
        if symbol in processed_symbols: continue

        ticker = f"{symbol}.BO"
        info = live_stock_data.get(ticker)
        if info and info.get('marketCap'):
            # BSE symbols are not in AMFI list, so we always use live data for categorization
            category = get_market_cap_category(info['marketCap'])
            
            final_json_data.append({
                "symbol": symbol,
                "company_name": info.get('longName', row['SC_NAME'].strip()),
                "exchange": "BSE",
                "market_cap_inr": info['marketCap'],
                "category": category
            })
            processed_symbols.add(symbol)

    # --- Write to JSON file ---
    with open('categorized_indian_stocks.json', 'w') as f:
        json.dump(final_json_data, f, indent=4)

    print(f"\nSuccessfully generated categorized_indian_stocks.json with {len(final_json_data)} total entries.")

if __name__ == '__main__':
    generate_stock_json()
